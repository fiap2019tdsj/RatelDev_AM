create table T_RD_USUARIO(
id_usuario integer primary key ,
nm_usuario varchar(100),
ds_email varchar(100),
ds_senha varchar (100)
);


create table T_RD_LOG_WATSON(
id_log_watson integer primary key,
ds_setenca varchar(100),
id_usuario integer references T_RD_USUARIO (id_usuario),
dt_data date
);



create table T_RD_MENU(
id_menu integer primary key,
nm_nome varchar(100),
ds_url varchar(100),
ds_descricao varchar(100)
);



create table T_RD_LOG(
id_log integer primary key,
ds_local varchar(100),
ds_metodo varchar(100),
ds_descricao varchar(100)
);
